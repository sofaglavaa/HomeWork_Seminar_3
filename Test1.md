## Вот моя инструкция

========================================================

git reflog - посмотреть историю

git reset --hard HEAD@{1} - выбрать нужный

git push --force - зафиксировать изменения на удаленном репозитории

========================================================

git config --list = просмотр конфигурации

git init  = создает пустой репозиторий в текущем репозитории



git status  = определяет текущее состояние нашего репозитория

git status -s = короткая версия  ( ??-новые неотслеживаемые файлы, А -отслеживаемые помечены, М-отредактированные файлы)



git remote add = добавляет удаленный источник данных (пример записи:  git remote add origin https://git)

git remote - v = список подключенных соединений



git add = добавление файла в индекс ( git add file.py )



git rm --cached file.py = удалить файл из индекса

rm -rf .git = удаление репозитория



git restore (название файла) = отменить на состояние предыдущего коммита

git restore -- staged (название файла) = убрать файл из индекса



git commit =  создание комита  ( i - для записи сообщения, ESC , w q - для выхода )

git commit -m ‘ сообщение ’  = короткая форма записи коммита

git commit -a = это тоже самое что и git add название файла и git commit

git commit -am ‘ сообщение ‘ = это тоже самое что и git add название файла и git commit + сообщение



.git ignore = запись файла дает возможность добавлять в него теги для игнорирования индексации

.git keep = создание данного файла в пустых дирректориях дает возможность показывать их для индекса



git log = получение информации о состоянии репозитория

git log --oneline = упрощенная информации о состоянии репозитория



git reset (хеш комита на который надо вернуться) = отмена комита и возвращение на нужный комит (откат)

git reflog = История всех изменений которые были совершены в работе с репозиторием

git revert = Не Отменяет а Сбрасывает оставляя все в истории в этом ее и отличеие от git reset



git branch -a = какие ветки есть в репозитории

git branch -b имя ветки = удаление ветки

git branch -D имя ветки = удаление ветки не смержив то есть без слияния

git branch (имя новой ветки) = создание новой ветки

git branch (имя новой ветки) (имя ветки с которой нужно создавать новую ветку ) = создание новой ветки от нужной точки



git checkout (имя ветки) = переключение на ветку

git checkout -b (имя новой ветки) = данная команда создаст новую ветку (имя новой ветки) и переключит на НЕЁ



git clone (сылка от куда клонируем репозиторий) (название дериктории куда клонировать) = клонирование удаленного ремозитория



git push = загрузка изменений в удаленный репозиторий

git push - u oridgin (название ветки) = загрузка изменений с локального сервира в удаленный с созданием отсутствующей ветки



git pull = загрузка изменений с удаленного репозитория на Локальный



git diff = просмотр какие были изменения

git diff  -- staged = просмотр какие были изменения уже в индексе (то есть после команды add)
